```dockerfile
    wireguard-go wireguard-tools \
```

```python
"""
Integration test: full init -> add-client -> verify WireGuard up -> remove-client -> verify gone.
Requires Docker with NET_ADMIN capability. Excluded from default pytest run.
Run with: pytest -m integration
"""
import subprocess
import pytest
from pathlib import Path
from click.testing import CliRunner
from wg_automate.main import cli  # adjust to actual CLI entry point

pytestmark = pytest.mark.integration

WIREGUARD_GO_IMAGE = "masipcat/wireguard-go@sha256:latest"
# NOTE: Pin to a specific digest before production use (Research open question 2).
# For now, use :latest and document the risk. Pinned digest: update manually during maintenance.

@pytest.fixture(scope="module")
def wg_interface(tmp_path_factory):
    """
    Set up a WireGuard interface using wireguard-go userspace implementation.
    Requires NET_ADMIN capability (provided by Docker --cap-add NET_ADMIN in CI).
    """
    wg_dir = tmp_path_factory.mktemp("wg_integration")

    # Skip if wireguard-go is not available (e.g., running on Windows CI)
    result = subprocess.run(["which", "wireguard-go"], capture_output=True)
    if result.returncode != 0:
        pytest.skip("wireguard-go not available in this environment")

    yield wg_dir

    # Cleanup: bring down the interface if it was brought up
    subprocess.run(["ip", "link", "delete", "wg0"], capture_output=True)


@pytest.fixture(scope="module")
def vault_dir(tmp_path_factory):
    """Isolated vault directory for the integration test session."""
    return tmp_path_factory.mktemp("vault_integration")


def run_cli(args, input_text=None, env=None):
    """Helper: run the CLI via CliRunner and return the result."""
    runner = CliRunner(mix_stderr=False)
    result = runner.invoke(cli, args, input=input_text, env=env, catch_exceptions=False)
    return result


@pytest.mark.integration
def test_full_lifecycle(wg_interface, vault_dir):
    """
    TEST-02: Full init -> add-client -> verify WireGuard interface up
    -> remove-client -> verify peer gone.

    This test drives the wg-automate CLI end-to-end inside a controlled environment.
    Requires wireguard-go and root/NET_ADMIN capability.
    """
    passphrase_input = "correct-horse-battery-staple-integration\n"

    # Step 1: init
    result = run_cli(
        ["--vault-dir", str(vault_dir), "init",
         "--endpoint", "10.99.0.1",
         "--subnet", "10.99.0.0/24",
         "--port", "51821"],
        input_text=passphrase_input + passphrase_input,  # confirm passphrase
    )
    assert result.exit_code == 0, f"init failed:\n{result.output}"

    # Step 2: add-client
    result = run_cli(
        ["--vault-dir", str(vault_dir), "add-client", "test-client"],
        input_text=passphrase_input,
    )
    assert result.exit_code == 0, f"add-client failed:\n{result.output}"

    # Step 3: verify peer appears in config
    config_path = vault_dir / "wg0.conf"
    if config_path.exists():
        config_content = config_path.read_text()
        assert "[Peer]" in config_content, "Peer section missing from config after add-client"

    # Step 4: remove-client
    result = run_cli(
        ["--vault-dir", str(vault_dir), "remove-client", "test-client"],
        input_text=passphrase_input,
    )
    assert result.exit_code == 0, f"remove-client failed:\n{result.output}"

    # Step 5: verify peer is gone from config
    if config_path.exists():
        config_content = config_path.read_text()
        # After removal the config should not contain the test-client peer
        # (exact verification depends on client name appearing in comments or AllowedIPs)
        assert "test-client" not in config_content or "[Peer]" not in config_content, \
            "Peer still present in config after remove-client"
```
