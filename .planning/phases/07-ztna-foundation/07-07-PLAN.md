MASTER_KEY = bytes(range(32))  # deterministic 32-byte test key
PASSPHRASE = bytearray(b"correct-horse-battery-staple")
WRONG_PASS = bytearray(b"wrong-passphrase")


def make_slot(admin_id: str = "owner", passphrase: bytearray = PASSPHRASE) -> Keyslot:
    return create_keyslot(admin_id, bytearray(passphrase), MASTER_KEY, _params=_DEV_FAST_PARAMS)


class TestKeyslotRoundTrip:
    def test_unlock_returns_master_key(self):
        slot = make_slot()
        recovered = unlock_keyslot(slot, bytearray(PASSPHRASE))
        assert bytes(recovered) == MASTER_KEY

    def test_wrong_passphrase_raises(self):
        slot = make_slot()
        with pytest.raises(KeyslotNotFoundError):
            unlock_keyslot(slot, bytearray(WRONG_PASS))

    def test_different_admins_same_master_key(self):
        slot_a = make_slot("alice", bytearray(b"alice-pass"))
        slot_b = make_slot("bob",   bytearray(b"bob-pass"))
        key_a = unlock_keyslot(slot_a, bytearray(b"alice-pass"))
        key_b = unlock_keyslot(slot_b, bytearray(b"bob-pass"))
        assert bytes(key_a) == MASTER_KEY
        assert bytes(key_b) == MASTER_KEY

    def test_wrong_passphrase_for_correct_admin(self):
        slot = make_slot("alice", bytearray(b"alice-pass"))
        with pytest.raises(KeyslotNotFoundError):
            unlock_keyslot(slot, bytearray(b"not-alice-pass"))


class TestKeyslotSerialization:
    def test_serialize_is_144_bytes(self):
        slot = make_slot()
        data = serialize_keyslot(slot)
        assert len(data) == 144, f"Expected 144 bytes, got {len(data)}"

    def test_deserialize_round_trip(self):
        slot = make_slot("owner")
        data = serialize_keyslot(slot)
        restored = deserialize_keyslot(data)
        assert restored.admin_id == slot.admin_id
        assert restored.argon2_salt == slot.argon2_salt
        assert restored.wrapping_iv == slot.wrapping_iv
        assert restored.wrapped_key == slot.wrapped_key
        assert restored.argon2_time == slot.argon2_time
        assert restored.argon2_mem == slot.argon2_mem
        assert restored.argon2_par == slot.argon2_par

        slot = make_slot()
        data = serialize_keyslot(slot)
        restored = deserialize_keyslot(data)
        recovered = unlock_keyslot(restored, bytearray(PASSPHRASE))

class TestTotpModule:
    def test_no_external_imports(self):
        """totp.py must not import pyotp or any non-stdlib package."""
        import importlib.util
        spec = importlib.util.find_spec("wireseal.security.totp")
        import ast, pathlib
        src = pathlib.Path(spec.origin).read_text()
        tree = ast.parse(src)
        for node in ast.walk(tree):
            if isinstance(node, (ast.Import, ast.ImportFrom)):
                names = [a.name for a in getattr(node, "names", [])]
                mod = getattr(node, "module", "") or ""
                for n in names + [mod]:
                    assert not n.startswith("pyotp"), "pyotp import found — forbidden"

    def test_secret_length(self, secret):
        assert len(secret) == 20

    def test_b32_round_trip(self, secret):
        b32 = secret_to_b32(secret)
        assert b32_to_secret(b32) == secret

    def test_uri_format(self, secret):
        uri = totp_uri(secret, "alice", "WireSeal")
        assert uri.startswith("otpauth://totp/")
        assert "alice" in uri
        assert "WireSeal" in uri


class TestTotpWindow:
    def test_current_code_accepted(self, secret):
        import hmac, struct, hashlib
        # Generate the current TOTP code manually (T=now//30)
        T = int(time.time()) // 30
        msg = struct.pack(">Q", T)
        h = hmac.new(secret, msg, hashlib.sha1).digest()
        offset = h[-1] & 0x0F
        code = struct.unpack(">I", h[offset:offset+4])[0] & 0x7FFFFFFF
        code_str = str(code % 10**6).zfill(6)
        assert verify_totp(secret, code_str, window=1)

    def test_wrong_code_rejected(self, secret):
        assert not verify_totp(secret, "000000", window=1)

    def test_wrong_length_rejected(self, secret):
        assert not verify_totp(secret, "12345", window=1)   # 5 digits
        assert not verify_totp(secret, "1234567", window=1)  # 7 digits

    def test_anti_replay_same_code_rejected_twice(self, secret):
        """Calling verify_totp twice with same code+used_set must reject second call."""
        import hmac, struct, hashlib
        T = int(time.time()) // 30
        msg = struct.pack(">Q", T)
        h = hmac.new(secret, msg, hashlib.sha1).digest()
        offset = h[-1] & 0x0F
        code = struct.unpack(">I", h[offset:offset+4])[0] & 0x7FFFFFFF
        code_str = str(code % 10**6).zfill(6)

        used: set[str] = set()
        assert verify_totp(secret, code_str, window=1, used_codes=used)
        # After first use, code is in 'used' — second call must fail
        assert not verify_totp(secret, code_str, window=1, used_codes=used)

        hashed = [hash_backup_code(c) for c in codes]
        assert verify_backup_code("XXXXXXXX", hashed) is None

    def test_backup_code_single_use(self):
        """Simulates single-use: remove hash after first verify, second must fail."""
        codes = generate_backup_codes(n=2)
        hashed = [hash_backup_code(c) for c in codes]

                hash_backup_code(c) for c in generate_backup_codes()

        msg = struct.pack(">Q", T)

        code_str = str(code % 10**6).zfill(6)

        assert verify_totp(secret, code_str, window=1)
