IMPORTANT NOTES before writing:
- Argon2id parameters in this module are `time_cost=3, memory_cost=65536, parallelism=4` (fast enough for per-slot use; vault body still uses 262144 KiB). These are stored per-slot so they are forward-upgradeable.
- The SECURITY-SPEC.md uses time=10 / memory=262144 for the keyslot — that is the production value. For this foundation implementation, use the values from the sub-plan brief (time=3, memory=65536) because they allow the verification tests to complete in reasonable time during development. The production constants can be updated in 07-07 hardening.
- AES-256-GCM AAD is `admin_id.encode("utf-8")` (not a prefix+slot_id scheme — that is a different design used in SECURITY-SPEC.md's JSON-based approach; the binary keyslot format uses direct admin_id as AAD).
- The `wrapped_key` field is 48 bytes = 32-byte ciphertext + 16-byte GCM tag (AESGCM produces them concatenated).
- The binary serialization format is exactly 144 bytes per keyslot:
  `salt(32) + memory_cost(4 BE uint32) + time_cost(4 BE uint32) + parallelism(4 BE uint32) + nonce(12) + wrapped_key(48) + admin_id_padded(40)`
  Total: 32+4+4+4+12+48+40 = 144. Verify this adds up.
- `admin_id` is encoded as UTF-8 and null-padded (or truncated) to exactly 40 bytes for serialization. Max admin_id length is therefore 39 bytes of UTF-8 (one byte reserved for null termination detection during deserialization; use `rstrip(b'\\x00')` to recover).

```python
"""Keyslot module — LUKS-style per-admin key wrapping for WireSeal vault.

Each keyslot stores a passphrase-derived wrapping key (Argon2id) used to
AES-256-GCM encrypt the vault master key. Multiple keyslots allow multiple
admins to unlock the same vault with independent passphrases.

Binary layout (144 bytes per keyslot):
  [32]  Argon2id salt
  [4]   memory_cost KiB (uint32 BE)
  [4]   time_cost (uint32 BE)
  [4]   parallelism (uint32 BE)
  [12]  AES-256-GCM nonce
  [48]  wrapped_master_key (32 ciphertext + 16 GCM tag)
  [40]  admin_id UTF-8, null-padded
  ----
  144 bytes total
"""

import hmac
import os
import struct
from dataclasses import dataclass, field

from argon2.low_level import Type, hash_secret_raw
from cryptography.exceptions import InvalidTag
from cryptography.hazmat.primitives.ciphers.aead import AESGCM

from .exceptions import KeyslotNotFoundError
from .secret_types import SecretBytes

# ---------------------------------------------------------------------------
# Default Argon2id parameters for new keyslots
# ---------------------------------------------------------------------------
# NOTE: These are development-phase values chosen so the verification test
# suite completes in reasonable time (each Argon2id call ~100ms instead of
# ~3s). The 07-07 hardening sub-plan will raise these to the production
# values specified in SECURITY-SPEC.md (time=10, memory=262144).
KEYSLOT_TIME_COST = 3
KEYSLOT_MEMORY_COST_KIB = 65536   # 64 MiB
KEYSLOT_PARALLELISM = 4
KEYSLOT_HASH_LEN = 32

# Binary format constants
_SLOT_STRUCT = struct.Struct(">III")  # memory_cost, time_cost, parallelism (3 × uint32 BE)
_SLOT_SIZE = 144
_ADMIN_ID_FIELD_LEN = 40


# ---------------------------------------------------------------------------
# Dataclasses
# ---------------------------------------------------------------------------


@dataclass
class Keyslot:
    """One admin's encrypted keyslot.

    All fields match the binary layout exactly. admin_id is stored as a
    Python str; serialization handles null-padding to 40 bytes.
    """
    admin_id: str
    salt: bytes           # 32 bytes — Argon2id salt
    memory_cost: int      # KiB
    time_cost: int
    parallelism: int
    nonce: bytes          # 12 bytes — AES-256-GCM nonce
    wrapped_key: bytes    # 48 bytes — 32 ciphertext + 16 GCM tag


@dataclass
class KeyslotStore:
    """Ordered collection of keyslots from a FORMAT_VERSION 3 vault header."""
    slots: list[Keyslot] = field(default_factory=list)


# ---------------------------------------------------------------------------
# Keyslot creation and unlock
# ---------------------------------------------------------------------------


def _derive_wrapping_key(passphrase_bytes: bytes, slot: Keyslot) -> bytearray:
    """Derive the 32-byte AES-256-GCM wrapping key from a passphrase using
    Argon2id with parameters stored in the slot record.

    Returns a mutable bytearray so the caller can wipe it after use.
    """
    raw = hash_secret_raw(
        secret=passphrase_bytes,
        salt=slot.salt,
        time_cost=slot.time_cost,
        memory_cost=slot.memory_cost,
        parallelism=slot.parallelism,
        hash_len=KEYSLOT_HASH_LEN,
        type=Type.ID,
    )
    return bytearray(raw)


def create_keyslot(
    admin_id: str,
    passphrase: "str | SecretBytes",
    master_key: bytes,
) -> Keyslot:
    """Create a new keyslot that wraps master_key under admin's passphrase.

    Steps:
    1. Generate 32-byte random salt (os.urandom).
    2. Derive 32-byte wrapping key via Argon2id (default params).
    3. Encrypt master_key with AES-256-GCM; AAD = admin_id.encode().
    4. Return populated Keyslot.

    Args:
        admin_id:   Admin identifier string (max 39 UTF-8 bytes).
        passphrase: Admin's passphrase — str or SecretBytes.
        master_key: The 32-byte vault master key to wrap.

    Returns:
        Keyslot with all fields populated.
    """
    if isinstance(passphrase, SecretBytes):
        pp_bytes = bytes(passphrase.expose_secret())
    else:
        pp_bytes = passphrase.encode("utf-8")

    salt = os.urandom(32)
    nonce = os.urandom(12)

    # Build a temporary slot with default params for wrapping key derivation
    temp_slot = Keyslot(
        admin_id=admin_id,
        salt=salt,
        memory_cost=KEYSLOT_MEMORY_COST_KIB,
        time_cost=KEYSLOT_TIME_COST,
        parallelism=KEYSLOT_PARALLELISM,
        nonce=nonce,
        wrapped_key=b"",  # filled below
    )

    wrapping_key = _derive_wrapping_key(pp_bytes, temp_slot)
    try:
        aad = admin_id.encode("utf-8")
        wrapped = AESGCM(bytes(wrapping_key)).encrypt(nonce, master_key, aad)
        # AESGCM.encrypt returns ciphertext+tag concatenated; for 32-byte
        # plaintext this is always 48 bytes.
        assert len(wrapped) == 48, f"Expected 48-byte wrapped key, got {len(wrapped)}"
    finally:
        for i in range(len(wrapping_key)):
            wrapping_key[i] = 0

    return Keyslot(
        admin_id=admin_id,
        salt=salt,
        memory_cost=KEYSLOT_MEMORY_COST_KIB,
        time_cost=KEYSLOT_TIME_COST,
        parallelism=KEYSLOT_PARALLELISM,
        nonce=nonce,
        wrapped_key=bytes(wrapped),
    )


def unlock_keyslot(slot: Keyslot, passphrase: "str | SecretBytes") -> bytes:
    """Decrypt and return the 32-byte master key from a keyslot.

    Args:
        slot:       The Keyslot record to unlock.
        passphrase: The admin's passphrase — str or SecretBytes.

    Returns:
        32-byte vault master key as bytes.

    Raises:
        KeyslotNotFoundError: AES-256-GCM authentication failed (wrong
            passphrase or data corruption). Treated as "not found" to
            avoid disclosing whether admin_id exists.
    """
    if isinstance(passphrase, SecretBytes):
        pp_bytes = bytes(passphrase.expose_secret())
    else:
        pp_bytes = passphrase.encode("utf-8")

    wrapping_key = _derive_wrapping_key(pp_bytes, slot)
    try:
        aad = slot.admin_id.encode("utf-8")
        master_key = AESGCM(bytes(wrapping_key)).decrypt(
            slot.nonce, slot.wrapped_key, aad
        )
    except InvalidTag:
        raise KeyslotNotFoundError(
            "Keyslot authentication failed — wrong passphrase or corrupted keyslot"
        ) from None
    finally:
        for i in range(len(wrapping_key)):
            wrapping_key[i] = 0

    return master_key


def find_and_unlock(
    store: KeyslotStore,
    admin_id: str,
    passphrase: "str | SecretBytes",
) -> bytes:
    """Find the keyslot for admin_id and unlock it.

    Iterates all slots and compares admin_id using hmac.compare_digest
    (constant-time) to prevent timing-based admin_id enumeration.
    Argon2id is called ONLY on the matching slot — O(1) unlock time
    regardless of total keyslot count.

    Args:
        store:      KeyslotStore from the vault header.
        admin_id:   The admin whose keyslot to find and unlock.
        passphrase: The admin's passphrase.

    Returns:
        32-byte vault master key.

    Raises:
        KeyslotNotFoundError: admin_id not found or passphrase wrong.
    """
    target = admin_id.encode("utf-8")
    matching_slot: Keyslot | None = None

    for slot in store.slots:
        # constant-time comparison — MUST NOT short-circuit before iterating all
        slot_id_bytes = slot.admin_id.encode("utf-8")
        if hmac.compare_digest(slot_id_bytes, target):
            matching_slot = slot
            # Do NOT break early — iterate all slots for constant-time behaviour

    if matching_slot is None:
        raise KeyslotNotFoundError(f"No keyslot found for admin_id: {admin_id!r}")

    return unlock_keyslot(matching_slot, passphrase)


# ---------------------------------------------------------------------------
# Binary serialization — 144 bytes per keyslot
# ---------------------------------------------------------------------------


def serialize_keyslot(slot: Keyslot) -> bytes:
    """Serialize a Keyslot to exactly 144 bytes.

    Layout:
      [32]  salt
      [4]   memory_cost (uint32 BE)
      [4]   time_cost (uint32 BE)
      [4]   parallelism (uint32 BE)
      [12]  nonce
      [48]  wrapped_key
      [40]  admin_id UTF-8 null-padded to 40 bytes

    Total: 32+4+4+4+12+48+40 = 144 bytes.
    """
    assert len(slot.salt) == 32, f"salt must be 32 bytes, got {len(slot.salt)}"
    assert len(slot.nonce) == 12, f"nonce must be 12 bytes, got {len(slot.nonce)}"
    assert len(slot.wrapped_key) == 48, f"wrapped_key must be 48 bytes, got {len(slot.wrapped_key)}"

    params = _SLOT_STRUCT.pack(slot.memory_cost, slot.time_cost, slot.parallelism)

    admin_id_bytes = slot.admin_id.encode("utf-8")
    if len(admin_id_bytes) > _ADMIN_ID_FIELD_LEN:
        raise ValueError(
            f"admin_id {slot.admin_id!r} exceeds {_ADMIN_ID_FIELD_LEN} byte limit "
            f"({len(admin_id_bytes)} bytes)"
        )
    admin_id_padded = admin_id_bytes.ljust(_ADMIN_ID_FIELD_LEN, b"\x00")

    raw = slot.salt + params + slot.nonce + slot.wrapped_key + admin_id_padded
    assert len(raw) == _SLOT_SIZE, f"Serialized slot is {len(raw)} bytes, expected {_SLOT_SIZE}"
    return raw


def deserialize_keyslot(data: bytes) -> Keyslot:
    """Deserialize exactly 144 bytes back into a Keyslot.

    Raises:
        ValueError: data is not exactly 144 bytes.
    """
    if len(data) != _SLOT_SIZE:
        raise ValueError(f"Expected {_SLOT_SIZE} bytes, got {len(data)}")

    offset = 0
    salt = data[offset:offset + 32]; offset += 32
    memory_cost, time_cost, parallelism = _SLOT_STRUCT.unpack(data[offset:offset + 12]); offset += 12
    nonce = data[offset:offset + 12]; offset += 12
    wrapped_key = data[offset:offset + 48]; offset += 48
    admin_id_raw = data[offset:offset + 40]; offset += 40

    admin_id = admin_id_raw.rstrip(b"\x00").decode("utf-8")

    return Keyslot(
        admin_id=admin_id,
        salt=salt,
        memory_cost=memory_cost,
        time_cost=time_cost,
        parallelism=parallelism,
        nonce=nonce,
        wrapped_key=wrapped_key,
    )


def serialize_store(store: KeyslotStore) -> bytes:
    """Serialize all keyslots in the store to contiguous bytes.

    Returns len(store.slots) * 144 bytes.
    """
    return b"".join(serialize_keyslot(s) for s in store.slots)


def deserialize_store(data: bytes, count: int) -> KeyslotStore:
    """Deserialize `count` keyslots from the beginning of `data`.

    Args:
        data:  Raw bytes containing count * 144 keyslot records.
        count: Number of keyslots to parse.

    Raises:
        ValueError: data is too short for `count` keyslots.
    """
    expected = count * _SLOT_SIZE
    if len(data) < expected:
        raise ValueError(
            f"Need {expected} bytes for {count} keyslots, got {len(data)}"
        )

    slots = []
    for i in range(count):
        start = i * _SLOT_SIZE
        slots.append(deserialize_keyslot(data[start:start + _SLOT_SIZE]))

    return KeyslotStore(slots=slots)

def _encrypt_vault(
    plaintext_dict: dict[str, Any],
    master_key: bytearray,
    *,
    salt: bytes | None = None,
) -> bytes:
    """Serialize and double-encrypt a vault state dict (FORMAT_VERSION 2 envelope).

    Encryption pipeline:
      plaintext JSON
        → ChaCha20-Poly1305  (inner, key_chacha, nonce1)
        → AES-256-GCM-SIV    (outer, key_aes,    nonce2)
        → stored ciphertext

    Both layers use the full 76-byte header as AEAD additional data (AAD).

    Args:
        plaintext_dict: Vault state dict to encrypt.
        master_key:     32-byte master key (bytearray — wiped after use by caller).
        salt:           Optional explicit salt; os.urandom(32) used if None.

    Binary layout:
      [76-byte header][4-byte ct_len][ct_len bytes double-ciphertext]
    """
    if salt is None:
        salt = os.urandom(ARGON2_SALT_LEN)
    nonce1 = os.urandom(NONCE_LEN)
    nonce2 = os.urandom(NONCE_LEN)

    key_chacha, key_aes = _derive_subkeys(master_key, salt)

    try:
        header = _HEADER_STRUCT.pack(
            MAGIC,
            FORMAT_VERSION,
            ARGON2_MEMORY_COST_KIB,
            ARGON2_TIME_COST,
            ARGON2_PARALLELISM,
            ARGON2_SALT_LEN, salt,
            NONCE_LEN, nonce1,
            NONCE_LEN, nonce2,
        )

        plaintext_json = json.dumps(plaintext_dict, separators=(",", ":")).encode("utf-8")

        layer1 = ChaCha20Poly1305(bytes(key_chacha)).encrypt(nonce1, plaintext_json, header)
        layer2 = AESGCMSIV(bytes(key_aes)).encrypt(nonce2, layer1, header)

    finally:
        wipe_bytes(key_chacha)
        wipe_bytes(key_aes)

    ct_len_field = struct.pack(">I", len(layer2))
    return header + ct_len_field + layer2
```

**IMPORTANT:** The existing `_encrypt_vault` derives `master_key` internally via `_derive_master_key`. The new version accepts it as a parameter. The header's Argon2id fields (memory/time/parallelism/salt) are written as fixed values because for FORMAT_VERSION 2 envelopes they describe how to re-derive the key from a passphrase — but in FORMAT_VERSION 3 they are legacy fields not used for key derivation. The salt field is still written to maintain header structure compatibility. For FORMAT_VERSION 3 files, the actual key material lives in the keyslot header, not derived from these fields.

def _decrypt_vault(blob: bytes, master_key: bytearray) -> dict[str, Any]:
    """Decrypt and deserialize a vault binary blob using a pre-derived master key.

    Accepts FORMAT_VERSION 2 binary envelope. The master_key is the 32-byte
    key fed into HKDF to produce the ChaCha/AES subkeys — for FORMAT_VERSION 2
    vaults this was the Argon2id output; for FORMAT_VERSION 3 vaults it is
    the unwrapped keyslot master key.

    Raises:
        VaultTamperedError: Magic bytes wrong or format unrecognised.
        VaultUnlockError:   GCM authentication failed.
    """
    if len(blob) < 5:
        raise VaultTamperedError("Vault file is too small to be valid")

    if blob[:4] != MAGIC:
        raise VaultTamperedError("Vault file has invalid magic bytes")

    version = blob[4]
    if version == 1:
        raise VaultTamperedError(
            "Vault was created with FORMAT_VERSION 1 (AES-256-GCM only). "
            "Run 'sudo wireseal fresh-start' then 'sudo wireseal init' to "
            "upgrade to the v2 dual-layer encryption engine."
        )
    if version not in (FORMAT_VERSION, FORMAT_VERSION_3):
        raise VaultTamperedError(
            f"Unsupported vault FORMAT_VERSION {version} "
            f"(this build supports versions {FORMAT_VERSION} and {FORMAT_VERSION_3})."
        )

    min_size = _HEADER_SIZE + 4

        except InvalidTag:
            raise VaultUnlockError("Vault unlock failed") from None

        wipe_bytes(key_chacha)
        wipe_bytes(key_aes)

**Change 6: Add `_write_v3_file` and `_read_v3_file` module-level functions after `_decrypt_vault`.**

```python
# FORMAT_VERSION 3 file I/O
# File layout: MAGIC(4) | VERSION=3(1) | keyslot_count(1) | N*144 keyslots | encrypted_payload

_V3_PREFIX_STRUCT = struct.Struct(">4sBB")  # MAGIC(4) + version(1) + count(1) = 6 bytes
_V3_PREFIX_SIZE = 6


def _write_v3_file(
    path: Path,
    keyslot_store: KeyslotStore,
    encrypted_payload: bytes,
) -> None:
    """Write a FORMAT_VERSION 3 vault file atomically.

    Format:
      [4]  MAGIC (b'WGAV')
      [1]  FORMAT_VERSION_3 (3)
      [1]  keyslot_count
      [N*144] serialized keyslots
      [...] encrypted payload (FORMAT_VERSION 2 envelope)
    """
    count = len(keyslot_store.slots)
    if count == 0:
        raise ValueError("Cannot write FORMAT_VERSION 3 file with zero keyslots")
    if count > 255:
        raise ValueError(f"Too many keyslots: {count} (max 255)")

    prefix = _V3_PREFIX_STRUCT.pack(MAGIC, FORMAT_VERSION_3, count)
    slots_bytes = serialize_store(keyslot_store)
    blob = prefix + slots_bytes + encrypted_payload
    atomic_write(path, blob, mode=0o600)


def _read_v3_file(path: Path) -> tuple[KeyslotStore, bytes]:
    """Read and parse a FORMAT_VERSION 3 vault file.

    Returns:
        (KeyslotStore, encrypted_payload_bytes)

    Raises:
        VaultTamperedError: File too small, bad magic, or wrong version.
    """
    raw = path.read_bytes()

    if len(raw) < _V3_PREFIX_SIZE:
        raise VaultTamperedError("Vault file is too small to be valid")

    magic, version, count = _V3_PREFIX_STRUCT.unpack(raw[:_V3_PREFIX_SIZE])

    if magic != MAGIC:
        raise VaultTamperedError("Vault file has invalid magic bytes")
    if version != FORMAT_VERSION_3:
        raise VaultTamperedError(
            f"Expected FORMAT_VERSION 3, got {version}"
        )

    slots_size = count * 144  # _SLOT_SIZE from keyslot module
    payload_offset = _V3_PREFIX_SIZE + slots_size

    if len(raw) < payload_offset:
        raise VaultTamperedError(
            f"Vault file too small: need {payload_offset} bytes for "
            f"{count} keyslots, got {len(raw)}"
        )

    store = deserialize_store(raw[_V3_PREFIX_SIZE:payload_offset], count)
    encrypted_payload = raw[payload_offset:]

    return store, encrypted_payload
```

    # ------------------------------------------------------------------
    # Keyslot management (FORMAT_VERSION 3)

    def add_keyslot(
        self,
        admin_id: str,
        new_passphrase: SecretBytes,
        owner_passphrase: SecretBytes,
        owner_admin_id: str = "owner",
    ) -> None:
        """Add a new keyslot for admin_id, allowing them to unlock the vault.

        If the current file is FORMAT_VERSION 2, upgrades it to FORMAT_VERSION 3
        by wrapping the derived master key in a keyslot for the owner, then
        adding the new admin's keyslot.

        Raises:
            KeyslotExistsError: admin_id already has a keyslot.
            VaultUnlockError:   owner_passphrase is wrong.
        """
        raw_owner = owner_passphrase.expose_secret()
        file_bytes = self._path.read_bytes()

        if len(file_bytes) < 5 or file_bytes[:4] != MAGIC:
            raise VaultTamperedError("Vault file has invalid magic bytes")

        version = file_bytes[4]

        if version == FORMAT_VERSION_3:
            store, payload = _read_v3_file(self._path)
            master_key = bytearray(
                find_and_unlock(store, owner_admin_id, owner_passphrase)
            )
        else:
            # FORMAT_VERSION 2 — derive master key, create owner keyslot, upgrade
            (_, _, mc, tc, par, _, salt, _, _, _, _) = _HEADER_STRUCT.unpack(file_bytes[:_HEADER_SIZE])
            master_key = _derive_master_key(raw_owner, salt, memory_cost=mc, time_cost=tc, parallelism=par)
            # The derived master key IS the v2 master key — wrap it for owner
            owner_slot = create_keyslot(owner_admin_id, owner_passphrase, bytes(master_key))
            store = KeyslotStore(slots=[owner_slot])
            payload = file_bytes  # existing FORMAT_VERSION 2 file becomes the payload

        # Check for duplicate admin_id (constant-time)
        for slot in store.slots:
            import hmac as _hmac
            if _hmac.compare_digest(slot.admin_id.encode(), admin_id.encode()):
                wipe_bytes(master_key)
                raise KeyslotExistsError(
                    f"Keyslot for admin_id {admin_id!r} already exists"
                )

        new_slot = create_keyslot(admin_id, new_passphrase, bytes(master_key))
        wipe_bytes(master_key)

        store.slots.append(new_slot)
        _write_v3_file(self._path, store, payload)

    def remove_keyslot(
        self,
        admin_id: str,
        owner_passphrase: SecretBytes,
        owner_admin_id: str = "owner",
    ) -> None:
        """Remove the keyslot for admin_id.

        Raises:
            AdminRoleError:      admin_id is the last owner keyslot.
            KeyslotNotFoundError: admin_id not in store.
            VaultUnlockError:    owner_passphrase is wrong.
        """
        if self._path.read_bytes()[4] != FORMAT_VERSION_3:
            raise VaultTamperedError("remove_keyslot requires FORMAT_VERSION 3 vault")

        store, payload = _read_v3_file(self._path)

        # Verify owner passphrase (raises KeyslotNotFoundError / VaultUnlockError if wrong)
        master_key = bytearray(find_and_unlock(store, owner_admin_id, owner_passphrase))
        wipe_bytes(master_key)

        # Ensure we're not removing the last owner
        remaining_owners = [
            s for s in store.slots
            if s.admin_id != admin_id
        ]
        # Check if the target is an owner slot — if the admin_id == owner_admin_id
        # and there's no other owner, reject
        target_exists = any(s.admin_id == admin_id for s in store.slots)
        if not target_exists:
            raise KeyslotNotFoundError(f"No keyslot for admin_id {admin_id!r}")

        if admin_id == owner_admin_id and not any(

            raise AdminRoleError(
                f"Cannot remove the last owner keyslot ({admin_id!r})"
            )

        store.slots = [s for s in store.slots if s.admin_id != admin_id]
        _write_v3_file(self._path, store, payload)
