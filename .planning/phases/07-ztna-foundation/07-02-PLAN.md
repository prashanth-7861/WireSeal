def _h_add_admin(self):
    if not _require_unlocked(self._send_json_error):
        return
    if not _require_owner(self._send_json_error):
        return
    body = self._read_json()
    admin_id = body.get("admin_id", "").strip()
    passphrase = body.get("passphrase", "")
    role = body.get("role", "admin")
    if not admin_id or not passphrase:
        self._send_json_error(400, {"error": "admin_id and passphrase required"})
        return
    if role not in ("owner", "admin", "readonly"):
        self._send_json_error(400, {"error": "role must be owner, admin, or readonly"})
        return
    vault = _session["vault"]
    passphrase_bytes = bytearray(passphrase.encode())
    try:
        vault.add_keyslot(admin_id, passphrase_bytes)
    except Exception as exc:
        self._send_json_error(409, {"error": str(exc)})
        return
    finally:
        passphrase_bytes[:] = b"\x00" * len(passphrase_bytes)
    # Add to admins dict in vault data
    with vault.open(_session["passphrase"], admin_id=_session.get("admin_id", "owner")) as state:
        state.data.setdefault("admins", {})[admin_id] = {
            "role": role,
            "created_at": _utcnow_iso(),
            "totp_secret_b32": None,
            "totp_enrolled_at": None,
            "backup_codes": [],
            "last_unlock": None,
        }
        _session["cache"]["admins"] = dict(state.data["admins"])
    audit_log.log("add-admin", {"target": admin_id, "role": role, "actor": _session.get("admin_id", "owner")})
    self._send_json(201, {"ok": True, "admin_id": admin_id})
```

def _h_remove_admin(self, target_id: str):
    if not _require_unlocked(self._send_json_error):
        return
    if not _require_owner(self._send_json_error):
        return
    if target_id == _session.get("admin_id"):
        self._send_json_error(409, {"error": "cannot remove yourself"})
        return
    vault = _session["vault"]
    admins = _session["cache"].get("admins", {})
    owners = [aid for aid, info in admins.items() if info.get("role") == "owner"]
    if target_id in owners and len(owners) == 1:
        self._send_json_error(409, {"error": "cannot remove the last owner"})
        return
    try:
        vault.remove_keyslot(target_id)
    except Exception as exc:
        self._send_json_error(404, {"error": str(exc)})
        return
    with vault.open(_session["passphrase"], admin_id=_session.get("admin_id", "owner")) as state:
        state.data.get("admins", {}).pop(target_id, None)
        _session["cache"]["admins"] = dict(state.data.get("admins", {}))
    audit_log.log("remove-admin", {"target": target_id, "actor": _session.get("admin_id", "owner")})
    self._send_json(200, {"ok": True})

def _h_change_admin_passphrase(self, target_id: str):
    if not _require_unlocked(self._send_json_error):
        return
    acting = _session.get("admin_id", "owner")
    acting_role = _session.get("admin_role", "owner")
    # Non-owner may only change their own passphrase
    if acting_role != "owner" and acting != target_id:
        self._send_json_error(403, {"error": "may only change your own passphrase"})
        return
    body = self._read_json()
    new_passphrase = body.get("new_passphrase", "")
    if not new_passphrase:
        self._send_json_error(400, {"error": "new_passphrase required"})
        return
    new_bytes = bytearray(new_passphrase.encode())
    try:
        _session["vault"].change_keyslot_passphrase(target_id, new_bytes)
    except Exception as exc:
        self._send_json_error(400, {"error": str(exc)})
        return
    finally:
        new_bytes[:] = b"\x00" * len(new_bytes)
    audit_log.log("change-passphrase", {"target": target_id, "actor": acting})
    self._send_json(200, {"ok": True})
```

              show_default=True, help="Role for the new admin")
def add_admin(admin_id: str, role: str):
    """Add a new admin keyslot to the vault."""
    vault_path = _vault_path()
    owner_pass = click.prompt("Owner passphrase", hide_input=True)
    new_pass = click.prompt(f"Passphrase for {admin_id!r}", hide_input=True)

    from wireseal.security.audit import AuditLog
    vault = Vault(vault_path)
    owner_bytes = bytearray(owner_pass.encode())
    new_bytes = bytearray(new_pass.encode())
    try:
        with vault.open(owner_bytes) as state:
            vault.add_keyslot(admin_id, new_bytes)
            state.data.setdefault("admins", {})[admin_id] = {
                "role": role_internal,
                "created_at": _utcnow_iso(),
                "totp_secret_b32": None,
                "totp_enrolled_at": None,
                "backup_codes": [],
                "last_unlock": None,

        audit = AuditLog(_audit_path())
        audit.log("add-admin", {"target": admin_id, "role": role_internal, "actor": "owner"})
        click.echo(f"Admin {admin_id!r} added with role {role_internal!r}.")
    except Exception as exc:
        click.echo(f"Error: {exc}", err=True)
        raise SystemExit(1)

        owner_bytes[:] = b"\x00" * len(owner_bytes)

import React, { useEffect, useState } from "react";
import { api } from "../api";
import { AdminRoleBadge } from "../components/AdminRoleBadge";

interface AdminInfo {
  id: string;
  role: string;
  totp_enrolled: boolean;
  last_unlock: string | null;
}

export function Admins() {
  const [admins, setAdmins] = useState<AdminInfo[]>([]);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState<string | null>(null);

  // Add form state
  const [newId, setNewId] = useState("");
  const [newPass, setNewPass] = useState("");
  const [newRole, setNewRole] = useState<"admin" | "owner" | "readonly">("admin");
  const [addError, setAddError] = useState<string | null>(null);
  const [adding, setAdding] = useState(false);

  const load = () => {
    setLoading(true);
    api.listAdmins()
      .then(data => { setAdmins(data.admins); setError(null); })
      .catch(err => setError(err.message))
      .finally(() => setLoading(false));
  };

  useEffect(() => { load(); }, []);

  const handleAdd = async (e: React.FormEvent) => {
    e.preventDefault();
    setAdding(true);
    setAddError(null);
    try {
      await api.addAdmin(newId, newPass, newRole);
      setNewId(""); setNewPass(""); setNewRole("admin");
      load();
    } catch (err: any) {
      setAddError(err.message ?? "Failed to add admin");
    } finally {
      setAdding(false);
    }
  };

  const handleRemove = async (targetId: string) => {
    if (!window.confirm(`Remove admin "${targetId}"? This cannot be undone.`)) return;
    try {
      await api.removeAdmin(targetId);
      load();
    } catch (err: any) {
      alert(err.message ?? "Failed to remove admin");
    }
  };

  // Determine current admin id from stored session (api exposes it or falls back)
  const currentAdminId = (api as any)._currentAdminId?.() ?? "owner";
  const ownerCount = admins.filter(a => a.role === "owner").length;

  return (
    <div className="p-6 space-y-8">
      <h1 className="text-2xl font-bold text-gray-900">Admins</h1>

      {loading && <p className="text-gray-500">Loading…</p>}
      {error && <p className="text-red-600">{error}</p>}

      {!loading && !error && (
        <table className="w-full text-sm border-collapse">
          <thead>
            <tr className="bg-gray-50 text-left text-gray-600 uppercase text-xs">
              <th className="px-4 py-2 border-b">Admin ID</th>
              <th className="px-4 py-2 border-b">Role</th>
              <th className="px-4 py-2 border-b">TOTP</th>
              <th className="px-4 py-2 border-b">Last Unlock</th>
              <th className="px-4 py-2 border-b"></th>
            </tr>
          </thead>
          <tbody>
            {admins.map(admin => {
              const isLastOwner = admin.role === "owner" && ownerCount === 1;
              const isSelf = admin.id === currentAdminId;
              const canRemove = !isLastOwner && !isSelf;
              return (
                <tr key={admin.id} className="hover:bg-gray-50">
                  <td className="px-4 py-2 border-b font-mono">{admin.id}</td>
                  <td className="px-4 py-2 border-b"><AdminRoleBadge role={admin.role} /></td>
                  <td className="px-4 py-2 border-b">{admin.totp_enrolled ? "Enrolled" : "—"}</td>
                  <td className="px-4 py-2 border-b text-gray-500">{admin.last_unlock ?? "Never"}</td>
                  <td className="px-4 py-2 border-b">
                    <button
                      onClick={() => handleRemove(admin.id)}
                      disabled={!canRemove}
                      title={isLastOwner ? "Cannot remove last owner" : isSelf ? "Cannot remove yourself" : ""}
                      className="text-red-600 hover:text-red-800 disabled:opacity-30 disabled:cursor-not-allowed text-xs"
                    >
                      Remove
                    </button>
                  </td>
                </tr>
              );
            })}
          </tbody>
        </table>
      )}

      <div className="border-t pt-6">
        <h2 className="text-lg font-semibold text-gray-800 mb-4">Add Admin</h2>
        <form onSubmit={handleAdd} className="space-y-3 max-w-sm">
          <div>
            <label className="block text-sm font-medium text-gray-700 mb-1">Admin ID</label>
            <input
              value={newId} onChange={e => setNewId(e.target.value)}
              className="w-full border rounded px-3 py-1.5 text-sm"
              placeholder="alice" required
            />
          </div>
          <div>
            <label className="block text-sm font-medium text-gray-700 mb-1">Passphrase</label>
            <input
              type="password" value={newPass} onChange={e => setNewPass(e.target.value)}
              className="w-full border rounded px-3 py-1.5 text-sm"
              required
            />
          </div>
          <div>
            <label className="block text-sm font-medium text-gray-700 mb-1">Role</label>
            <select
              value={newRole} onChange={e => setNewRole(e.target.value as any)}
              className="w-full border rounded px-3 py-1.5 text-sm"
            >
              <option value="admin">Admin</option>
              <option value="owner">Owner</option>
              <option value="readonly">Read-only</option>
            </select>
          </div>
          {addError && <p className="text-red-600 text-sm">{addError}</p>}
          <button
            type="submit" disabled={adding}
            className="bg-blue-600 text-white px-4 py-2 rounded text-sm hover:bg-blue-700 disabled:opacity-50"
          >
            {adding ? "Adding…" : "Add Admin"}
          </button>
        </form>
      </div>
    </div>
  );
}
