"""TOTP (RFC 6238) implementation — stdlib only, no external dependencies."""
from __future__ import annotations
import base64
import hashlib
import hmac
import os
import secrets
import struct
import time


def generate_totp_secret() -> bytes:
    """Generate a 20-byte random TOTP secret (160 bits, standard for TOTP)."""
    return os.urandom(20)


def totp_uri(secret: bytes, admin_id: str, issuer: str = "WireSeal") -> str:
    """Generate otpauth:// URI for QR code enrollment.

    Compatible with Google Authenticator, Aegis, Bitwarden, etc.
    """
    from urllib.parse import quote
    secret_b32 = base64.b32encode(secret).decode("ascii")
    label = quote(f"{issuer}:{admin_id}")
    params = f"secret={secret_b32}&issuer={quote(issuer)}&algorithm=SHA1&digits=6&period=30"
    return f"otpauth://totp/{label}?{params}"


def _hotp(secret: bytes, counter: int) -> int:
    """RFC 4226 HOTP: HMAC-SHA1 based OTP for given counter value."""
    msg = struct.pack(">Q", counter)
    h = hmac.new(secret, msg, hashlib.sha1).digest()
    offset = h[-1] & 0x0F
    code = struct.unpack(">I", h[offset:offset + 4])[0] & 0x7FFFFFFF
    return code % 1_000_000


def verify_totp(secret: bytes, code: str, window: int = 2) -> bool:
    """Verify a 6-digit TOTP code.

    Checks current window ± `window` steps (default ±2 = ±60s tolerance for clock drift).
    Returns True if any window matches. Does NOT consume the code (caller handles replay).

    Args:
        secret: Raw 20-byte secret (not base32-encoded)
        code: 6-character digit string e.g. "123456"
        window: Number of 30s steps before/after current to check
    """
    if not code or not code.isdigit() or len(code) != 6:
        return False
    t = int(time.time()) // 30
    for delta in range(-window, window + 1):
        if hmac.compare_digest(str(_hotp(secret, t + delta)).zfill(6), code.zfill(6)):
            return True
    return False


def generate_backup_codes(n: int = 8) -> list[str]:
    """Generate n single-use backup codes.

    Each code is 8 uppercase alphanumeric characters (~47 bits entropy).
    Format: "XXXXXXXX" (8 chars, no dashes — easy to type).
    """
    alphabet = "ABCDEFGHJKLMNPQRSTUVWXYZ23456789"  # Unambiguous: no 0/O, 1/I
    return ["".join(secrets.choice(alphabet) for _ in range(8)) for _ in range(n)]


def hash_backup_code(code: str) -> str:
    """SHA-256 hex digest of an uppercase backup code. Stored in vault."""
    return hashlib.sha256(code.upper().encode()).hexdigest()


def verify_backup_code(code: str, hashed_codes: list[str]) -> str | None:
    """Check if code matches any stored hash. Returns the matched hash (for removal) or None."""
    code_hash = hash_backup_code(code.upper())
    for h in hashed_codes:
        if hmac.compare_digest(code_hash, h):
            return h
    return None


def secret_to_b32(secret: bytes) -> str:
    """Encode raw secret bytes to base32 string for storage/display."""
    return base64.b32encode(secret).decode("ascii")


def b32_to_secret(b32: str) -> bytes:
    """Decode base32 string to raw secret bytes."""
    return base64.b32decode(b32.upper())

def _h_unlock(req: "_Handler", _groups: tuple) -> dict:
    client_ip = req.client_address[0]
    _check_rate_limit(client_ip)

    body           = req._json()
    passphrase_str = body.get("passphrase", "")
    admin_id       = body.get("admin_id", "owner")
    if not passphrase_str:
        raise _ApiError("passphrase is required", 400)

    from wireseal.security.secret_types import SecretBytes
    from wireseal.security.secrets_wipe import wipe_string
    from wireseal.security.vault        import Vault
    from wireseal.security.audit        import AuditLog

    passphrase = SecretBytes(bytearray(passphrase_str.encode()))
    try:
        vault = Vault(_VAULT_PATH)
        try:
            with vault.open(passphrase) as st:
                cache = _refresh_cache(st)
                # Read admin TOTP state from vault while we have it open
                admins_data = st.data.get("admins", {}) if hasattr(st, "data") else {}
                admin_entry = admins_data.get(admin_id, {})
                totp_secret_b32 = admin_entry.get("totp_secret_b32")
        except Exception:
            passphrase.wipe()
            _record_unlock_failure(client_ip)
            raise _ApiError("Incorrect passphrase.", 401)

        # TOTP enforcement — only when admin has TOTP enrolled
        if totp_secret_b32:
            import time as _time
            totp_code = body.get("totp_code", "")
            from wireseal.security.totp import verify_totp, b32_to_secret
            secret = b32_to_secret(totp_secret_b32)
            if not totp_code:
                raise _ApiError("TOTP code required.", 401, {"totp_required": True})
            # Anti-replay: reject codes already accepted in this 30s window
            t_step = str(int(_time.time()) // 30)
            replay_key = f"{t_step}:{totp_code}"
            if replay_key in _totp_used_codes:
                raise _ApiError("TOTP code already used.", 401)
            if not verify_totp(secret, totp_code):
                raise _ApiError("Invalid TOTP code.", 401)
            _totp_used_codes.add(replay_key)
            # Prune entries older than 3 time steps to bound set growth
            current_step = int(_time.time()) // 30
            stale = {k for k in _totp_used_codes if k.split(":")[0].isdigit() and int(k.split(":")[0]) < current_step - 3}
            _totp_used_codes.difference_update(stale)

        with _lock:
            if _session["passphrase"]:
                _session["passphrase"].wipe()
            _session.update(vault=vault, passphrase=passphrase, cache=cache,
                            admin_id=admin_id, admin_role=admin_entry.get("role", "owner"))

        _clear_unlock_failures(client_ip)
        AuditLog(_AUDIT_PATH).log("unlock-web", {"admin_id": admin_id})

        # Auto-start WireGuard tunnel if config exists but tunnel is down
        try:
            wg_check = subprocess.run(
                _sudo(["wg", "show", _WG_IFACE]),
                capture_output=True, timeout=5,
                creationflags=_SP_FLAGS,
            )
            if wg_check.returncode != 0:
                conf_path = Path("/etc/wireguard") / f"{_WG_IFACE}.conf"
                if sys.platform == "win32":
                    conf_path = Path(os.environ.get("PROGRAMDATA", r"C:\ProgramData")) / "WireGuard" / f"{_WG_IFACE}.conf"
                if conf_path.exists():
                    subprocess.run(
                        _sudo(["wg-quick", "up", _WG_IFACE]),
                        capture_output=True, timeout=15,
                        creationflags=_SP_FLAGS,
                    )
        except Exception:
            pass

        return {"ok": True, "role": admin_entry.get("role", "owner")}
    finally:
        wipe_string(passphrase_str)

export function TotpEnrollDialog({ onClose }: Props) {

    setError("");

      const res = await api.totpEnrollConfirm(code);

      setLoading(false);

            <X className="w-5 h-5" />
          </button>
