# Plan 08-02: File Manager API

## Goal
Expose a configurable share directory via REST with strict path traversal
prevention. Upload/download work with large files without loading them into
server memory. All write operations audit-logged.

- `src/wireseal/api.py` — 8 new routes + `_resolve_safe_path` + vault config key

## Tasks

### Task 1 — `_resolve_safe_path`

```python
import os as _os
from pathlib import Path as _Path

def _resolve_safe_path(share_root: str, user_path: str) -> _Path:
    """Resolve user_path relative to share_root.

    Raises _ApiError(403) on any path traversal attempt.
    share_root must be an absolute, resolved directory.
    """
    root = _Path(share_root).resolve()
    # Strip leading slashes so Path join doesn't treat it as absolute
    sanitized = user_path.lstrip("/\\")
    target = (root / sanitized).resolve()
    if not str(target).startswith(str(root) + _os.sep) and target != root:
        raise _ApiError(403, "path outside share root")
    return target
```

### Task 2 — File Manager config endpoints

```
GET  /api/files/config   → _handle_fm_get_config
PUT  /api/files/config   → _handle_fm_put_config
```

**GET** returns `{"share_root": "...", "max_upload_mb": 100, "enabled": false}`.
Does not require admin (read is safe).

**PUT** body: `{"share_root": "/path/to/shared", "max_upload_mb": 100, "enabled": true}`
- Requires owner or admin role
- Validates `share_root` is absolute, exists, is directory
- Rejects share roots that are system paths (`/etc`, `/bin`, `C:\Windows`, etc.)
  via `_is_system_path(root)` allowlist check
- Saves to vault; broadcasts `"files_config"` SSE event

```python
def _handle_fm_list(self) -> None:
    _require_unlocked(self)
    root, _ = _fm_config(self)       # raises 503 if disabled / not configured
    q = _parse_qs(self.path).get("path", [""])[0]
    target = _resolve_safe_path(root, q)
    if not target.is_dir():
        raise _ApiError(404, "not a directory")
    entries = []
    for entry in sorted(target.iterdir(), key=lambda e: (not e.is_dir(), e.name.lower())):
        st = entry.stat(follow_symlinks=False)
        entries.append({
            "name": entry.name,
            "is_dir": entry.is_dir(),
            "size": st.st_size if not entry.is_dir() else None,
            "mtime": st.st_mtime,
        })
    _json_ok(self, {"path": q or "/", "entries": entries})
```

### Task 4 — File download (streaming)

```
GET /api/files/download?path=subdir/file.zip
```

Stream in 64 KB chunks — never load full file into memory:

```python
CHUNK = 65536

def _handle_fm_download(self) -> None:
    _require_unlocked(self)
    root, _ = _fm_config(self)
    q = _parse_qs(self.path).get("path", [""])[0]
    target = _resolve_safe_path(root, q)
    if not target.is_file():
        raise _ApiError(404, "not a file")
    size = target.stat(follow_symlinks=False).st_size
    name = _urllib.parse.quote(target.name)
    self.send_response(200)
    self.send_header("Content-Type", "application/octet-stream")
    self.send_header("Content-Disposition", f'attachment; filename*=UTF-8\'\'{name}')
    self.send_header("Content-Length", str(size))
    self.end_headers()
    with target.open("rb") as fh:
        while chunk := fh.read(CHUNK):
            self.wfile.write(chunk)
```

### Task 5 — File upload (streaming)

```
POST /api/files/upload?path=subdir/
```

Content-Type: multipart/form-data

Parse multipart manually using stdlib `cgi.FieldStorage` (already used
elsewhere in the server). Write to `.tmp` file then `os.replace()` atomically.

```python
def _handle_fm_upload(self) -> None:
    _require_unlocked(self)
    root, cfg = _fm_config(self)
    max_bytes = cfg.get("max_upload_mb", 100) * 1024 * 1024

    content_length = int(self.headers.get("Content-Length", 0))
    if content_length > max_bytes:
        raise _ApiError(413, f"upload exceeds {cfg['max_upload_mb']} MB limit")

    q = _parse_qs(self.path).get("path", [""])[0]
    dest_dir = _resolve_safe_path(root, q)
    if not dest_dir.is_dir():
        raise _ApiError(404, "destination directory does not exist")

    env = {"REQUEST_METHOD": "POST",
           "CONTENT_TYPE": self.headers.get("Content-Type", ""),
           "CONTENT_LENGTH": str(content_length)}
    form = _cgi.FieldStorage(fp=self.rfile, headers=self.headers,
                             environ=env, limit=max_bytes)
    uploaded = form["file"]
    filename = _Path(uploaded.filename).name  # strip any path component
    if not filename:
        raise _ApiError(400, "filename missing")

    dest_dir = _resolve_safe_path(root, q)  # re-validate after parsing
    tmp = dest_dir / (filename + ".tmp")
    final = dest_dir / filename

    _resolve_safe_path(root, str(tmp.relative_to(_Path(root).resolve())))

    try:
        with tmp.open("wb") as fh:
            while chunk := uploaded.file.read(CHUNK):
                fh.write(chunk)
        _os.replace(tmp, final)
    except Exception:
        tmp.unlink(missing_ok=True)
        raise

    _audit_write(f"file_upload path={final} size={final.stat().st_size}")
    _sse_broker.broadcast("files", {"event": "upload", "path": q, "name": filename})
    _json_ok(self, {"uploaded": filename})
```

```python
def _handle_fm_mkdir(self) -> None:
    _require_unlocked(self)
    root, _ = _fm_config(self)
    body = _read_json(self)
    path = body.get("path", "")
    target = _resolve_safe_path(root, path)
    if target.exists():
        raise _ApiError(409, "already exists")
    target.mkdir(parents=False)
    _set_permissions(target, 0o700)
    _audit_write(f"file_mkdir path={target}")
    _json_ok(self, {"created": path})
```

```python
def _handle_fm_delete(self) -> None:
    _require_unlocked(self)
    root, _ = _fm_config(self)
    q = _parse_qs(self.path).get("path", [""])[0]
    target = _resolve_safe_path(root, q)
    if not target.exists():
        raise _ApiError(404, "not found")
    if target == _Path(root).resolve():
        raise _ApiError(400, "cannot delete share root")
    if target.is_dir():
        try:
            target.rmdir()  # only empty dirs
        except OSError:
            raise _ApiError(409, "directory not empty")
    else:
        target.unlink()
    _audit_write(f"file_delete path={target}")
    _sse_broker.broadcast("files", {"event": "delete", "path": q})
    _json_ok(self, {"deleted": q})
```

```python
def _handle_fm_rename(self) -> None:
    _require_unlocked(self)
    root, _ = _fm_config(self)
    body = _read_json(self)
    src = _resolve_safe_path(root, body.get("from", ""))
    dst = _resolve_safe_path(root, body.get("to", ""))
    if not src.exists():
        raise _ApiError(404, "source not found")
    if dst.exists():
        raise _ApiError(409, "destination already exists")
    _os.replace(src, dst)
    _audit_write(f"file_rename from={src} to={dst}")
    _json_ok(self, {"from": body["from"], "to": body["to"]})
```

### Task 10 — `_fm_config` helper

```python
def _fm_config(handler) -> tuple[str, dict]:
    """Extract and validate filemanager config from vault. Raises 503 if disabled."""
    with _vault_ctx() as state:
        cfg = state.data.get("filemanager", {})
    if not cfg.get("enabled") or not cfg.get("share_root"):
        raise _ApiError(503, "file manager not configured")
    root = cfg["share_root"]
    resolved = str(_Path(root).resolve())
    if not _Path(resolved).is_dir():
        raise _ApiError(503, f"share root does not exist: {root}")
    return resolved, cfg
```

### Task 11 — `_is_system_path` guard

```python
_SYSTEM_PATHS = {
    "/etc", "/bin", "/sbin", "/usr", "/boot", "/proc", "/sys", "/dev",
    "C:\\Windows", "C:\\Program Files", "C:\\Program Files (x86)",
}

def _is_system_path(path: str) -> bool:
    p = str(_Path(path).resolve())
    return any(p == s or p.startswith(s + _os.sep) for s in _SYSTEM_PATHS)
```
