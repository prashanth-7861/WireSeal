### Task 1 — `_SSEBroker` in api.py

Add above the `_Handler` class:

```python
import queue as _queue

class _SSEBroker:
    """Thread-safe SSE fanout broker. One queue per connected client."""

    def __init__(self) -> None:
        self._lock = _threading.Lock()
        self._clients: dict[int, _queue.SimpleQueue] = {}
        self._next_id = 0

    def subscribe(self) -> tuple[int, _queue.SimpleQueue]:
        q: _queue.SimpleQueue = _queue.SimpleQueue()
        with self._lock:
            cid = self._next_id
            self._next_id += 1
            self._clients[cid] = q
        return cid, q

    def unsubscribe(self, cid: int) -> None:
        with self._lock:
            self._clients.pop(cid, None)

    def broadcast(self, event: str, data: object) -> None:
        payload = f"event: {event}\ndata: {_json.dumps(data)}\n\n"
        with self._lock:
            clients = list(self._clients.values())
        for q in clients:
            q.put(payload)

    def ping(self) -> None:
        self.broadcast("ping", {})


_sse_broker = _SSEBroker()
```

Handler:
```python
def _handle_sse(self) -> None:
    _require_unlocked(self)  # 401 if locked
    cid, q = _sse_broker.subscribe()
    self.send_response(200)
    self.send_header("Content-Type", "text/event-stream")
    self.send_header("Cache-Control", "no-cache")
    self.send_header("Connection", "keep-alive")
    self.send_header("X-Accel-Buffering", "no")  # nginx: disable proxy buffering
    self.end_headers()
    try:
        while True:
            try:
                chunk = q.get(timeout=30)
                self.wfile.write(chunk.encode())
                self.wfile.flush()
            except _queue.Empty:
                # send keepalive comment
                self.wfile.write(b": keepalive\n\n")
                self.wfile.flush()
    except (BrokenPipeError, ConnectionResetError):
        pass
    finally:
        _sse_broker.unsubscribe(cid)
```

### Task 4 — `useServerEvents` hook in Dashboard

**`Dashboard/src/app/hooks/useServerEvents.ts`** (new file):

```typescript
import { useEffect, useRef } from "react";
import { API_BASE } from "../api";

type EventHandlers = Partial<Record<string, (data: unknown) => void>>;

export function useServerEvents(handlers: EventHandlers): void {
  const handlersRef = useRef(handlers);
  handlersRef.current = handlers;

  useEffect(() => {
    let es: EventSource | null = null;
    let retryDelay = 1000;

    function connect() {
      es = new EventSource(`${API_BASE}/api/events`, { withCredentials: false });

      es.onopen = () => {
        retryDelay = 1000;
      };

      Object.keys(handlersRef.current).forEach((event) => {
        es!.addEventListener(event, (e: MessageEvent) => {
          try {
            const handler = handlersRef.current[event];
            if (handler) handler(JSON.parse(e.data));
          } catch {
            // malformed event — ignore
          }
        });
      });

      es.onerror = () => {
        es?.close();
        setTimeout(connect, retryDelay);
        retryDelay = Math.min(retryDelay * 2, 30_000);
      };
    }

    connect();
    return () => es?.close();
  }, []); // stable: handlers accessed via ref
}
```

### Task 8 — BroadcastChannel for vault lock (multi-tab)

```typescript
const _vaultChannel = new BroadcastChannel("wireseal-vault");

export function broadcastVaultLocked(): void {
  _vaultChannel.postMessage({ locked: true });
  window.dispatchEvent(new Event("vault-locked")); // same-tab compat
}

// In app root:
useEffect(() => {
  const ch = new BroadcastChannel("wireseal-vault");
  ch.onmessage = (e) => {
    if (e.data?.locked) handleVaultLocked();
  };
  return () => ch.close();
}, []);
```

## Notes
- `_build_status_payload()` must be extracted from the GET /api/status inline
  handler body into a standalone function so both the HTTP handler and the SSE
  tick can call it without code duplication.
- Do NOT stream large binary data through SSE. File downloads remain HTTP GET.
- The 5 s status tick broadcasts even with no connected SSE clients — cost is
  one `wg show` subprocess per 5 s, same as before (existing poll moved server-
  side). With clients connected, polling load drops to zero.
