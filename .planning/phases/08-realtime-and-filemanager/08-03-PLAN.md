### Task 1 — API client functions (api.ts)

```typescript
export interface FileEntry {
  name: string;
  is_dir: boolean;
  size: number | null;
  mtime: number;
}

export interface FmConfig {
  share_root: string;
  max_upload_mb: number;
  enabled: boolean;
}

export async function getFmConfig(): Promise<FmConfig> {
  return apiFetch("/api/files/config");
}

export async function setFmConfig(cfg: Partial<FmConfig>): Promise<void> {
  return apiFetch("/api/files/config", { method: "PUT", body: cfg });
}

export async function listFiles(path: string): Promise<{ path: string; entries: FileEntry[] }> {
  return apiFetch(`/api/files?path=${encodeURIComponent(path)}`);
}

export function downloadUrl(path: string): string {
  return `${API_BASE}/api/files/download?path=${encodeURIComponent(path)}`;
}

export async function deleteFile(path: string): Promise<void> {
  return apiFetch(`/api/files?path=${encodeURIComponent(path)}`, { method: "DELETE" });
}

export async function createDir(path: string): Promise<void> {
  return apiFetch("/api/files/mkdir", { method: "POST", body: { path } });
}

export async function renameFile(from: string, to: string): Promise<void> {
  return apiFetch("/api/files/rename", { method: "POST", body: { from, to } });
}

// Upload returns a Promise with progress via XHR (not fetch — fetch has no upload progress)
export function uploadFile(
  dirPath: string,
  file: File,
  onProgress: (pct: number) => void
): Promise<void> {
  return new Promise((resolve, reject) => {
    const fd = new FormData();
    fd.append("file", file);
    const xhr = new XMLHttpRequest();
    xhr.open("POST", `${API_BASE}/api/files/upload?path=${encodeURIComponent(dirPath)}`);
    xhr.upload.onprogress = (e) => {
      if (e.lengthComputable) onProgress(Math.round((e.loaded / e.total) * 100));
    };
    xhr.onload = () => (xhr.status < 300 ? resolve() : reject(new Error(xhr.responseText)));
    xhr.onerror = () => reject(new Error("upload failed"));
    xhr.send(fd);
  });
}
```

function Breadcrumbs({ path, onNavigate }: { path: string; onNavigate: (p: string) => void }) {
  const parts = path.split("/").filter(Boolean);
  return (
    <nav className="fm-breadcrumbs">
      <button onClick={() => onNavigate("")}>Home</button>
      {parts.map((part, i) => {
        const p = parts.slice(0, i + 1).join("/");
        return (
          <span key={p}>
            <span className="fm-sep">/</span>
            <button onClick={() => onNavigate(p)}>{part}</button>
          </span>
        );
      })}
    </nav>
  );
}

### Task 5 — Rename inline input

When `renaming === entry.name`, replace name cell with:

```tsx
<input
  autoFocus
  defaultValue={renaming}
  onKeyDown={async (ev) => {
    if (ev.key === "Enter") {
      const newName = ev.currentTarget.value.trim();
      if (newName && newName !== renaming) {
        await renameFile(join(currentPath, renaming), join(currentPath, newName));
        setRenaming(null);
        load(currentPath);
      }
    }
    if (ev.key === "Escape") setRenaming(null);
  }}
/>
```

### Task 6 — FileUploadZone (drag-drop)

```tsx
// FileUploadZone.tsx
export function FileUploadZone({ dirPath, onDone }: { dirPath: string; onDone: () => void }) {
  const [dragging, setDragging] = useState(false);

  async function handleFiles(files: FileList) {
    for (const file of Array.from(files)) {
      // UploadProgress renders progress per-file — managed in parent via state
      onStartUpload(file);
      await uploadFile(dirPath, file, (pct) => updateProgress(file.name, pct));
      onDone();
    }
  }

  return (
    <div
      className={`fm-dropzone ${dragging ? "dragging" : ""}`}
      onDragOver={(e) => { e.preventDefault(); setDragging(true); }}
      onDragLeave={() => setDragging(false)}
      onDrop={(e) => { e.preventDefault(); setDragging(false); handleFiles(e.dataTransfer.files); }}
    >
      <input type="file" multiple onChange={(e) => e.target.files && handleFiles(e.target.files)} />
      <span>Drop files here or <label>browse</label></span>
    </div>
  );
}
```
