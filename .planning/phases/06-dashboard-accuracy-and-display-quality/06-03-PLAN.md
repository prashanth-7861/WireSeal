```typescript
  useEffect(() => {
    let cancelled = false;
    const poll = async () => {
      try {
        const s = await api.status();
        if (!cancelled) setPeerStatus(s);
      } catch {
        // Status unavailable — keep previous peerStatus, don't clear it
      }
    };
    poll();
    const id = window.setInterval(poll, 5000);
    return () => { cancelled = true; clearInterval(id); };
  }, []);
```
